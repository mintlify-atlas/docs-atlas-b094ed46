## Overview

The Generations API allows you to create TTS audio from text and retrieve generation history. All operations require an active subscription.

## Create Generation

Generate speech audio from text using a voice clone.

```typescript
trpc.generations.create.useMutation();
```

<Warning>
  Requires an **active subscription**. Returns `FORBIDDEN` with message `SUBSCRIPTION_REQUIRED` if no active subscription exists.
</Warning>

### Input Schema

<ParamField path="text" type="string" required>
  Text to convert to speech. Must be between 1 and 5,000 characters.
</ParamField>

<ParamField path="voiceId" type="string" required>
  ID of the voice to use (can be SYSTEM or CUSTOM variant)
</ParamField>

<ParamField path="temperature" type="number" default="0.8">
  Controls randomness in generation. Range: 0.0 to 2.0
  
  - Lower values (0.0-0.5): More consistent, predictable output
  - Default (0.8): Balanced naturalness
  - Higher values (1.0-2.0): More varied, expressive output
</ParamField>

<ParamField path="topP" type="number" default="0.95">
  Nucleus sampling threshold. Range: 0.0 to 1.0
  
  Controls diversity by considering only the top probability mass.
</ParamField>

<ParamField path="topK" type="number" default="1000">
  Number of top tokens to consider. Range: 1 to 10,000
  
  Limits the sampling pool to the K most likely tokens.
</ParamField>

<ParamField path="repetitionPenalty" type="number" default="1.2">
  Penalty for repeating tokens. Range: 1.0 to 2.0
  
  - 1.0: No penalty (may lead to repetition)
  - 1.2: Default (balanced)
  - 2.0: Strong penalty (avoids repetition)
</ParamField>

### Response

<ResponseField name="id" type="string" required>
  Unique identifier for the created generation. Use this to retrieve the audio via `/api/audio/{id}`
</ResponseField>

### Implementation

<CodeGroup>
```typescript src/trpc/routers/generations.ts
create: orgProcedure
  .input(
    z.object({
      text: z.string().min(1).max(TEXT_MAX_LENGTH),
      voiceId: z.string().min(1),
      temperature: z.number().min(0).max(2).default(0.8),
      topP: z.number().min(0).max(1).default(0.95),
      topK: z.number().min(1).max(10000).default(1000),
      repetitionPenalty: z.number().min(1).max(2).default(1.2),
    })
  )
  .mutation(async ({ input, ctx }) => {
    // Check for active subscription
    const customerState = await polar.customers.getStateExternal({
      externalId: ctx.orgId,
    });
    const hasActiveSubscription =
      (customerState.activeSubscriptions ?? []).length > 0;
    if (!hasActiveSubscription) {
      throw new TRPCError({
        code: "FORBIDDEN",
        message: "SUBSCRIPTION_REQUIRED",
      });
    }

    // Validate voice access
    const voice = await prisma.voice.findUnique({
      where: {
        id: input.voiceId,
        OR: [
          { variant: "SYSTEM" },
          { variant: "CUSTOM", orgId: ctx.orgId }
        ],
      },
    });

    if (!voice?.r2ObjectKey) {
      throw new TRPCError({
        code: "NOT_FOUND",
        message: "Voice not found",
      });
    }

    // Generate audio via Chatterbox API
    const { data, error } = await chatterbox.POST("/generate", {
      body: {
        prompt: input.text,
        voice_key: voice.r2ObjectKey,
        temperature: input.temperature,
        top_p: input.topP,
        top_k: input.topK,
        repetition_penalty: input.repetitionPenalty,
        norm_loudness: true,
      },
      parseAs: "arrayBuffer",
    });

    // Store in database and R2
    const generation = await prisma.generation.create({
      data: {
        orgId: ctx.orgId,
        text: input.text,
        voiceName: voice.name,
        voiceId: voice.id,
        temperature: input.temperature,
        topP: input.topP,
        topK: input.topK,
        repetitionPenalty: input.repetitionPenalty,
      },
    });

    await uploadAudio({ 
      buffer: Buffer.from(data), 
      key: `generations/orgs/${ctx.orgId}/${generation.id}` 
    });

    // Track usage in Polar (fire-and-forget)
    polar.events.ingest({
      events: [{
        name: "tts_generation",
        externalCustomerId: ctx.orgId,
        metadata: { characters: input.text.length },
        timestamp: new Date(),
      }],
    }).catch(() => {});

    return { id: generation.id };
  }),
```

```typescript Example Usage
import { trpc } from '@/lib/trpc';

function GenerateButton() {
  const createGeneration = trpc.generations.create.useMutation({
    onSuccess: (data) => {
      console.log('Generation created:', data.id);
      // Audio available at `/api/audio/${data.id}`
    },
    onError: (error) => {
      if (error.message === 'SUBSCRIPTION_REQUIRED') {
        // Redirect to billing
      }
    },
  });
  
  const handleGenerate = () => {
    createGeneration.mutate({
      text: "Hello from Resonance!",
      voiceId: "voice-id-here",
      temperature: 0.8,
      topP: 0.95,
      topK: 1000,
      repetitionPenalty: 1.2,
    });
  };
  
  return (
    <button onClick={handleGenerate}>
      Generate Speech
    </button>
  );
}
```

## Get All Generations

Retrieve all generations for the current organization.

```typescript
trpc.generations.getAll.useQuery();
```

### Response

<ResponseField name="generations" type="Generation[]">
  Array of generation objects ordered by creation date (newest first)
  
  <Expandable title="Generation object">
    <ResponseField name="id" type="string" required>
      Unique generation identifier
    </ResponseField>
    
    <ResponseField name="text" type="string" required>
      Original text that was converted to speech
    </ResponseField>
    
    <ResponseField name="voiceId" type="string" required>
      ID of the voice used
    </ResponseField>
    
    <ResponseField name="voiceName" type="string" required>
      Display name of the voice
    </ResponseField>
    
    <ResponseField name="temperature" type="number" required>
      Temperature value used (0.0-2.0)
    </ResponseField>
    
    <ResponseField name="topP" type="number" required>
      Top-p value used (0.0-1.0)
    </ResponseField>
    
    <ResponseField name="topK" type="number" required>
      Top-k value used (1-10000)
    </ResponseField>
    
    <ResponseField name="repetitionPenalty" type="number" required>
      Repetition penalty used (1.0-2.0)
    </ResponseField>
    
    <ResponseField name="createdAt" type="Date" required>
      Timestamp when the generation was created
    </ResponseField>
  </Expandable>
</ResponseField>

### Implementation

<CodeGroup>
```typescript src/trpc/routers/generations.ts
getAll: orgProcedure.query(async ({ ctx }) => {
  const generations = await prisma.generation.findMany({
    where: { orgId: ctx.orgId },
    orderBy: { createdAt: "desc" },
    omit: {
      orgId: true,
      r2ObjectKey: true,
    },
  });

  return generations;
}),
```

```typescript Example Usage
import { trpc } from '@/lib/trpc';

function GenerationHistory() {
  const { data: generations, isLoading } = trpc.generations.getAll.useQuery();
  
  if (isLoading) return <div>Loading...</div>;
  
  return (
    <div>
      <h2>Generation History</h2>
      {generations?.map(gen => (
        <div key={gen.id}>
          <p>{gen.text}</p>
          <audio src={`/api/audio/${gen.id}`} controls />
          <span>Voice: {gen.voiceName}</span>
          <span>Created: {gen.createdAt.toLocaleDateString()}</span>
        </div>
      ))}
    </div>
  );
}
```
</CodeGroup>

## Get Generation by ID

Retrieve a specific generation with audio URL.

```typescript
trpc.generations.getById.useQuery({ id: "gen-123" });
```

### Input Schema

<ParamField path="id" type="string" required>
  Generation ID to retrieve
</ParamField>

### Response

<ResponseField name="generation" type="Generation">
  Generation object with all fields plus audioUrl
  
  <Expandable title="Extended Generation object">
    Contains all standard Generation fields plus:
    
    <ResponseField name="audioUrl" type="string" required>
      URL path to access the audio file: `/api/audio/{id}`
    </ResponseField>
  </Expandable>
</ResponseField>

### Implementation

<CodeGroup>
```typescript src/trpc/routers/generations.ts
getById: orgProcedure
  .input(z.object({ id: z.string() }))
  .query(async ({ input, ctx }) => {
    const generation = await prisma.generation.findUnique({
      where: { id: input.id, orgId: ctx.orgId },
      omit: {
        orgId: true,
        r2ObjectKey: true,
      },
    });

    if (!generation) {
      throw new TRPCError({ code: "NOT_FOUND" });
    }

    return {
      ...generation,
      audioUrl: `/api/audio/${generation.id}`,
    };
  }),
```

```typescript Example Usage
import { trpc } from '@/lib/trpc';

function GenerationPlayer({ generationId }: { generationId: string }) {
  const { data: generation } = trpc.generations.getById.useQuery({
    id: generationId,
  });
  
  if (!generation) return null;
  
  return (
    <div>
      <h3>{generation.voiceName}</h3>
      <p>{generation.text}</p>
      <audio src={generation.audioUrl} controls />
      <details>
        <summary>Parameters</summary>
        <ul>
          <li>Temperature: {generation.temperature}</li>
          <li>Top-p: {generation.topP}</li>
          <li>Top-k: {generation.topK}</li>
          <li>Repetition Penalty: {generation.repetitionPenalty}</li>
        </ul>
      </details>
    </div>
  );
}
```
</CodeGroup>

## Error Codes

| Code | Description | When It Occurs |
|------|-------------|----------------|
| `UNAUTHORIZED` | User not authenticated | Missing or invalid session |
| `FORBIDDEN` | Subscription required | No active subscription (message: `SUBSCRIPTION_REQUIRED`) |
| `NOT_FOUND` | Resource not found | Voice or generation doesn't exist or not accessible |
| `PRECONDITION_FAILED` | Voice audio unavailable | Voice exists but has no audio file |
| `INTERNAL_SERVER_ERROR` | Generation failed | Chatterbox API error or storage failure |

## Usage Tracking

Each successful generation triggers a usage event sent to Polar for billing:

```typescript
polar.events.ingest({
  events: [{
    name: "tts_generation",
    externalCustomerId: ctx.orgId,
    metadata: { characters: input.text.length },
    timestamp: new Date(),
  }],
});
```

- Event name: `tts_generation`
- Metered by character count
- Fire-and-forget (doesn't block response)
- Silent failure (doesn't affect user experience)
